// Swift code snippet
struct WorkoutTimer: View {
    @State private var isRunning = false
    @State private var timeRemaining = 300
    
    var body: some View {
        VStack {
            Text(timeString(from: timeRemaining))
                .font(.system(size: 54, weight: .light))
            Button(isRunning ? "Pause" : "Start") {
                isRunning.toggle()
            }
        }
    }
}
